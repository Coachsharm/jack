#!/bin/bash
# Auto-start monitors for Jack-John communication
# Called on boot via cron @reboot

LOG_DIR="/root/.openclaw/workspace"
TIMESTAMP=$(date '+%Y-%m-%d %H:%M:%S')

echo "[$TIMESTAMP] Starting BOT_CHAT monitors..." >> "$LOG_DIR/monitor-startup.log"

# Wait for system to be fully ready
sleep 10

# Start Jack's monitor (watches John's BOT_CHAT)
if ! pgrep -f "monitor-bot-chat.sh" > /dev/null; then
    nohup /root/.openclaw/workspace/monitor-bot-chat.sh > "$LOG_DIR/monitor-bot-chat.log" 2>&1 &
    echo "[$TIMESTAMP] Started Jack's monitor (PID: $!)" >> "$LOG_DIR/monitor-startup.log"
else
    echo "[$TIMESTAMP] Jack's monitor already running" >> "$LOG_DIR/monitor-startup.log"
fi

# Wait for Docker to be ready
sleep 5

# Start John's monitor (watches Jack's messages)
if docker exec openclaw-john pgrep -f "monitor-jack-chat.sh" > /dev/null 2>&1; then
    echo "[$TIMESTAMP] John's monitor already running" >> "$LOG_DIR/monitor-startup.log"
else
    docker exec openclaw-john bash -c 'nohup /home/openclaw/.openclaw/workspace/monitor-jack-chat.sh > /home/openclaw/.openclaw/workspace/monitor-jack-chat.log 2>&1 &'
    echo "[$TIMESTAMP] Started John's monitor" >> "$LOG_DIR/monitor-startup.log"
fi

echo "[$TIMESTAMP] Monitor startup complete" >> "$LOG_DIR/monitor-startup.log"

# Start BOT_CHAT Relay Bridge (John/Jack <-> Ross)
if ! pgrep -f 'bot-chat-relay.sh' > /dev/null; then
    nohup /root/openclaw-clients/bot-chat-relay.sh > /root/openclaw-clients/relay-bridge.log 2>&1 &
    echo "[$TIMESTAMP] Started BOT_CHAT relay bridge (PID: $!)" >> "$LOG_DIR/monitor-startup.log"
else
    echo "[$TIMESTAMP] Relay bridge already running" >> "$LOG_DIR/monitor-startup.log"
fi

#!/bin/bash
# Monitor John's BOT_CHAT.md for changes and trigger my heartbeat
WATCH_FILE="/root/openclaw-clients/john/workspace/BOT_CHAT.md"
LAST_HASH=""

while true; do
    if [ -f "$WATCH_FILE" ]; then
        CURRENT_HASH=$(md5sum "$WATCH_FILE" | awk '{print $1}')
        
        if [ -n "$LAST_HASH" ] && [ "$CURRENT_HASH" != "$LAST_HASH" ]; then
            # File changed! Trigger my heartbeat
            openclaw system event --text "BOT_CHAT updated" --mode now &
            echo "$(date): BOT_CHAT changed, triggered heartbeat"
        fi
        
        LAST_HASH="$CURRENT_HASH"
    fi
    
    sleep 2
done

===== 8. RELAY BRIDGE =====
#!/bin/bash
# =============================================================================
# BOT_CHAT Relay Bridge ‚Äî Host-Side (v3)
# =============================================================================
# PURPOSE: Bridges John/Jack's shared BOT_CHAT.md ‚Üê‚Üí Ross's separate BOT_CHAT.md
#
# ARCHITECTURE:
#   Jack's BOT_CHAT.md ‚Üí symlink ‚Üí John's BOT_CHAT.md (SHARED FILE)
#   Ross's BOT_CHAT.md ‚Üí separate file (ISOLATED)
#
# LOCATION: /root/openclaw-clients/bot-chat-relay.sh
# =============================================================================

JOHN_BOTCHAT="/root/openclaw-clients/john/workspace/BOT_CHAT.md"
ROSS_BOTCHAT="/root/openclaw-clients/ross/workspace/BOT_CHAT.md"
RELAY_LOG="/root/openclaw-clients/relay-bridge.log"
RELAY_STATE_DIR="/root/openclaw-clients/.relay-state"
JOHN_STATE="$RELAY_STATE_DIR/john-relayed-lines"
ROSS_STATE="$RELAY_STATE_DIR/ross-relayed-lines"

mkdir -p "$RELAY_STATE_DIR"

# Initialize state files to current line counts (skip existing content)
[ -f "$JOHN_STATE" ] || wc -l < "$JOHN_BOTCHAT" > "$JOHN_STATE" 2>/dev/null || echo "0" > "$JOHN_STATE"
[ -f "$ROSS_STATE" ] || wc -l < "$ROSS_BOTCHAT" > "$ROSS_STATE" 2>/dev/null || echo "0" > "$ROSS_STATE"

log() {
    echo "$(date '+%Y-%m-%d %H:%M:%S') | $1" >> "$RELAY_LOG"
}

# Match patterns: "‚Üí Ross", "-> Ross", "‚Üí ALL", "-> ALL", "- Ross", "- John", "- Jack", "- ALL"
# The header format is: "### TIMESTAMP SGT - Sender ‚Üí Target (#N)"
# We match on the marker line (###) that contains the target name
is_for_ross() {
    echo "$1" | grep -qiE '(Ross|ALL)' && echo "$1" | grep -q '###'
}

is_for_jack_or_john() {
    echo "$1" | grep -qiE '(Jack|John|ALL)' && echo "$1" | grep -q '###'
}

log "===== RELAY BRIDGE v3 STARTED ====="
log "John file: $JOHN_BOTCHAT"
log "Ross file: $ROSS_BOTCHAT"

LOOP_COUNT=0

while true; do
    # =============================================
    # John/Jack ‚Üí Ross
    # =============================================
    JOHN_LAST=$(cat "$JOHN_STATE" 2>/dev/null || echo "0")
    JOHN_CURRENT=$(wc -l < "$JOHN_BOTCHAT" 2>/dev/null || echo "0")
    
    if [ "$JOHN_CURRENT" -gt "$JOHN_LAST" ]; then
        NEW_LINES=$(tail -n "+$((JOHN_LAST + 1))" "$JOHN_BOTCHAT")
        
        # Check if any ### header line mentions Ross or ALL
        HEADER_LINES=$(echo "$NEW_LINES" | grep '###')
        MATCHES_ROSS=$(echo "$HEADER_LINES" | grep -ciE 'Ross|ALL' 2>/dev/null || echo "0")
        
        if [ "$MATCHES_ROSS" -gt 0 ]; then
            log "RELAY: John/Jack ‚Üí Ross detected (lines $JOHN_LAST ‚Üí $JOHN_CURRENT)"
            
            # Append to Ross's file
            printf "\n%s\n" "$NEW_LINES" >> "$ROSS_BOTCHAT"
            chown 101000:101000 "$ROSS_BOTCHAT" 2>/dev/null
            
            # Update Ross state to prevent echo-back
            ROSS_NOW=$(wc -l < "$ROSS_BOTCHAT" 2>/dev/null || echo "0")
            echo "$ROSS_NOW" > "$ROSS_STATE"
            
            # Wake Ross with EXPLICIT instructions
            docker exec openclaw-ross openclaw system event \
                --text "New BOT_CHAT message for you. FOLLOW HEARTBEAT.md NOW: 1) Read BOT_CHAT.md 2) Write your reply to BOT_CHAT.md 3) Post summary to Telegram group -5213725265 using message tool 4) Update .bot-chat-state. ALL STEPS MANDATORY." --mode now >> "$RELAY_LOG" 2>&1 &
            
            log "RELAY: Relayed to Ross + woke Ross"
        else
            log "SKIP: John file +$((JOHN_CURRENT - JOHN_LAST)) lines, not for Ross"
        fi
        
        echo "$JOHN_CURRENT" > "$JOHN_STATE"
    fi
    
    # =============================================
    # Ross ‚Üí Jack/John
    # =============================================
    ROSS_LAST=$(cat "$ROSS_STATE" 2>/dev/null || echo "0")
    ROSS_CURRENT=$(wc -l < "$ROSS_BOTCHAT" 2>/dev/null || echo "0")
    
    if [ "$ROSS_CURRENT" -gt "$ROSS_LAST" ]; then
        NEW_LINES=$(tail -n "+$((ROSS_LAST + 1))" "$ROSS_BOTCHAT")
        
        HEADER_LINES=$(echo "$NEW_LINES" | grep '###')
        MATCHES_JACK=$(echo "$HEADER_LINES" | grep -ciE 'Jack|John|ALL' 2>/dev/null || echo "0")
        
        if [ "$MATCHES_JACK" -gt 0 ]; then
            log "RELAY: Ross ‚Üí Jack/John detected (lines $ROSS_LAST ‚Üí $ROSS_CURRENT)"
            
            # Append to John's file (Jack reads via symlink)
            printf "\n%s\n" "$NEW_LINES" >> "$JOHN_BOTCHAT"
            chown 101000:101000 "$JOHN_BOTCHAT" 2>/dev/null
            
            # Update John state to prevent echo-back
            JOHN_NOW=$(wc -l < "$JOHN_BOTCHAT" 2>/dev/null || echo "0")
            echo "$JOHN_NOW" > "$JOHN_STATE"
            
            # Wake target(s)
            if echo "$HEADER_LINES" | grep -qiE 'Jack|ALL'; then
                openclaw system event \
                    --text "New BOT_CHAT message from Ross. FOLLOW HEARTBEAT.md: Read BOT_CHAT.md, write reply, post to Telegram -5213725265, update state. ALL STEPS MANDATORY." --mode now >> "$RELAY_LOG" 2>&1 &
                log "RELAY: Woke Jack (native)"
            fi
            if echo "$HEADER_LINES" | grep -qiE 'John|ALL'; then
                docker exec openclaw-john openclaw system event \
                    --text "New BOT_CHAT message from Ross. FOLLOW HEARTBEAT.md: Read BOT_CHAT.md, write reply, post to Telegram -5213725265, update state. ALL STEPS MANDATORY." --mode now >> "$RELAY_LOG" 2>&1 &
                log "RELAY: Woke John (Docker)"
            fi
            
            log "RELAY: Relayed to John/Jack"
        else
            log "SKIP: Ross file +$((ROSS_CURRENT - ROSS_LAST)) lines, not for Jack/John"
        fi
        
        echo "$ROSS_CURRENT" > "$ROSS_STATE"
    fi
    
    # Trim log every ~5 min
    LOOP_COUNT=$((LOOP_COUNT + 1))
    if [ $((LOOP_COUNT % 150)) -eq 0 ]; then
        if [ "$(wc -l < "$RELAY_LOG" 2>/dev/null)" -gt 500 ]; then
            tail -200 "$RELAY_LOG" > "$RELAY_LOG.tmp" && mv "$RELAY_LOG.tmp" "$RELAY_LOG"
        fi
    fi
    
    sleep 2
done
